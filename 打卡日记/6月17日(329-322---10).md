### [329. 矩阵中的最长递增路径](https://leetcode-cn.com/problems/longest-increasing-path-in-a-matrix/)

```java
class Solution {
    int n, m;
    int[] dx = {-1, 0, 1, 0}, dy = {0, 1, 0, -1};
    public int longestIncreasingPath(int[][] matrix) {
        n = matrix.length;
        m = matrix[0].length;
        if (matrix == null || n == 0 || m == 0) return 0;
        int[][] f = new int[n][m];
        int ans = 0;
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) ans = Math.max(ans, dp(i, j, f, matrix));
        }
        return ans;
    }

    public int dp(int x, int y, int[][] f, int[][] matrix) {
        if (f[x][y] != 0) return f[x][y];
        f[x][y] = 1;
        for (int i = 0; i < 4; ++i) {
            int a = x + dx[i], b = y + dy[i];
            if (a >= 0 && a < n && b >= 0 && b < m && matrix[x][y] < matrix[a][b]) f[x][y] = Math.max(f[x][y], dp(a, b, f, matrix) + 1);
        }
        return f[x][y];
    }
}
```
