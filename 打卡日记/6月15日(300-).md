### [300. 最长递增子序列](https://leetcode-cn.com/problems/longest-increasing-subsequence/)

```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length, ans = 0;
        int[] f = new int[n];
        f[0] = 1;
        for (int i = 0; i < n; ++i) {
            f[i] = 1;
            for (int j = 0; j < i; ++j) {
                if (nums[j] < nums[i]) f[i] = Math.max(f[i], f[j] + 1);
            }
            ans = Math.max(ans, f[i]);
        }
        return ans;
    }
}
```

### [301. 删除无效的括号](https://leetcode-cn.com/problems/remove-invalid-parentheses/)

* 左括号数目等于右括号数目；
* 任意前缀中左括号数目大于等于右括号数目。

```java
class Solution {
    List<String> ans = new ArrayList<>();
    public List<String> removeInvalidParentheses(String s) {
        int l = 0, r = 0;
        for (Character c: s.toCharArray()) {
            if (c == '(') l++;
            else if (c == ')') {
                if (l == 0) r++;
                else l--;
            }
        }
        dfs(s.toCharArray(), 0, "", 0, l, r);
        return ans;
    }
	
    // u：当前遍历到哪个字符
    // cnt：当前左括号减去右括号数量
    // l：当前还可以删去几个左括号
    // r：当前还可以删去几个右括号
    public void dfs(char[] s, int u, String path, int cnt, int l, int r) {
        if (u == s.length) {
            if (cnt == 0) ans.add(path);
            return;
        }
        if (s[u] != '(' && s[u] != ')') dfs(s, u + 1, path + s[u], cnt, l, r);
        else if (s[u] == '(') {
            int k = u;
            while (k < s.length && s[k] == '(') k++;
            l -= k - u;
            for (int i = k - u; i >= 0; --i) {
                if (l >= 0) dfs(s, k, path, cnt, l, r);
                path += '(';
                cnt++;
                l++;
            }
        } else if (s[u] == ')') {
            int k = u;
            while (k < s.length && s[k] == ')') k++;
            r -= k - u;
            for (int i = k - u; i >= 0; --i) {
                if (cnt >= 0 && r >= 0) dfs(s, k, path, cnt, l, r);
                path += ')';
                cnt--;
                r++; 
            }
        }
    }
}
```

### [303. 区域和检索 - 数组不可变](https://leetcode-cn.com/problems/range-sum-query-immutable/)

```java
class NumArray {
    int[] sum;
    public NumArray(int[] nums) {
        int n = nums.length;
        sum = new int[n + 1];
        for (int i = 0; i < n; ++i) sum[i + 1] = sum[i] + nums[i];
    }
    
    public int sumRange(int left, int right) {
        return sum[right + 1] - sum[left];
    }
}
```

### [304. 二维区域和检索 - 矩阵不可变](https://leetcode-cn.com/problems/range-sum-query-2d-immutable/)

```java
class NumMatrix {
    int[][] sum;
    public NumMatrix(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        sum = new int[n + 1][m + 1];
        for (int i = 1; i <= n; ++i) {
            for (int j = 1; j <= m; ++j) {
                sum[i][j] = sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1] + matrix[i - 1][j - 1];
            }
        }
    }
    
    public int sumRegion(int x1, int y1, int x2, int y2) {
        x1++; y1++; x2++; y2++;
        return sum[x2][y2] - sum[x2][y1 - 1] - sum[x1 - 1][y2] + sum[x1 - 1][y1 -1];
    }
}
```

### [306. 累加数](https://leetcode-cn.com/problems/additive-number/)

```java
class Solution {
    public boolean isAdditiveNumber(String num) {
        int n = num.length();
        for (int i = 0; i < n; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int a = -1, b = i, c = j;
                while (true) {
                    String s1 = num.substring(a + 1, b + 1);
                    String s2 = num.substring(b + 1, c + 1);
                    if (s1.length() > 1 && s1.charAt(0) == '0') break;
                    if (s2.length() > 1 && s2.charAt(0) == '0') break;
                    String s3 = add(s1, s2);
                    if (c + s3.length() + 1 <= n && num.substring(c + 1, s3.length() + c + 1).equals(s3)) {
                        a = b;
                        b = c;
                        c = c + s3.length();
                        if (c + 1 == n) return true;
                    } else break;
                }
            }
        }
        return false;
    }

    public String add(String s1, String s2) {
        s1 = new StringBuilder(s1).reverse().toString();
        s2 = new StringBuilder(s2).reverse().toString();
        String s3 = "";
        for (int i = 0, c = 0; i < s1.length() || i < s2.length() || c > 0; ++i) {
            if (i < s1.length()) c += s1.charAt(i) - '0';
            if (i < s2.length()) c += s2.charAt(i) - '0';
            s3 += c % 10;
            c /= 10;
        }
        return new StringBuilder(s3).reverse().toString();
    }
}
```

```java
class Solution {
    public boolean isAdditiveNumber(String num) {
        return dfs(num, 0, 0, 0, 0);
    }

    // u:当前下标
    // sum:前两个数之和
    // pre:前一个数的值
    // cnt:已经生成几个数
    public boolean dfs(String num, int u, long sum, long pre, int cnt) {
        int n = num.length();
        if (u == n) return cnt >= 3;
        long val = 0;
        for (int i = u; i < n; ++i) {
            if (i > u && num.charAt(u) == '0') break;
            val = val * 10 + num.charAt(i) - '0';
            if (cnt >= 2) {
                if (val < sum) continue;
                else if (val > sum) break;
            }
            if (dfs(num, i + 1, pre + val, val, cnt + 1)) return true;
        }
        return false;
    }
}
```
