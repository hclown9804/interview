### 思路

f[i, j]

* 所有测量区间长度是i且有j个鸡蛋的测量方案；

* 最坏情况下的最小值

$$
f[i,j]=min(f[i,j-1],max(f[k-1,j-1],f[i-k,j])+1)
$$

### 代码

```c++
#include <iostream>
using namespace std;

const int N = 110, M = 11;
int f[N][M], n, m;

int main() {
    while (cin >> n >> m) {
        for (int i = 1; i <= n; ++i) f[i][1] = i;
        for (int i = 1; i <= m; ++i) f[1][i] = 1;
        
        for (int i = 2; i <= n; ++i) {
            for (int j = 2; j <= m; ++j) {
                f[i][j] = f[i][j - 1];
                for (int k = 1; k <= i; ++k) 
                    f[i][j] = min(f[i][j], max(f[k - 1][j - 1], f[i - k][j]) + 1);
            }
        }
        cout << f[n][m] << endl;
    }
    return 0;
}
```
