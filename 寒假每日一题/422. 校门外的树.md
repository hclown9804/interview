# 代码

```c++
#include <iostream>
#include <algorithm>
using namespace std;

const int N = 10010;
int L, n;
bool st[N];

int main() {
    cin >> L >> n;
    for (int i = 0; i <= L; ++i) st[i] = true;
    while (n--) {
        int a, b;
        cin >> a >> b;
        for (int i = a; i <= b; ++i) st[i] = false;
    }
    int ans = 0;
    for (int i = 0; i <= L; ++i) ans += st[i];
    cout << ans << endl;
    return 0;
}
```
