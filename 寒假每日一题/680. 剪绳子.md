# 代码

```c++
#include <iostream>
#include <algorithm>
using namespace std;

const int N = 100010;
int n, m, a[N];

bool check(double mid) {
    int cnt = 0;
    for (int i = 0; i < n; ++i) {
        cnt += a[i] / mid;
    }
    return cnt >= m;
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; ++i) cin >> a[i];
    
    double l = 0, r = 1e9;
    while (r - l > 1e-4) {
        double mid = (l + r) / 2;
        if (check(mid)) l = mid;
        else r = mid;
    }
    printf("%.2f\n", r);
    return 0;
}
```
