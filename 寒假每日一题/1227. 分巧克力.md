# 代码

```c++
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
const int N = 100010;
int n, k, h[N], w[N];

bool check(int mid) {
    LL ans = 0;
    for (int i = 0; i < n; ++i) {
        ans += (LL)h[i] / mid * (w[i] / mid);
        if (ans >= k) return true;
    }
    return false;
}

int main() {
    cin >> n >> k;
    for (int i = 0; i < n; ++i) {
        cin >> h[i] >> w[i];
    }
    int l = 1, r = 100000;
    while (l < r) {
        int mid = (l + r + 1) >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    cout << r << endl;
    return 0;
}
```
