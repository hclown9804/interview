# 代码

```c++
#include <iostream>
#include <algorithm>
using namespace std;
const int N = 510;

int n, f[N][N], a[N][N];

int main() {
    cin >> n;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= i; ++j) {
            cin >> a[i][j];
        }
    }
    
    for (int i = 1; i <= n; ++i) {
        for (int j = 0; j <= i + 1; ++j) {
            f[i][j] = -0x3f3f3f3f;
        }
    }
    
    f[1][1] = a[1][1];
    for (int i = 2; i <= n; ++i) {
        for (int j = 1; j <= i; ++j) {
            f[i][j] = a[i][j] + max(f[i - 1][j - 1], f[i - 1][j]);
        }
    }
    int ans = -0x3f3f3f3f;
    for (int i = 1; i <= n; ++i) ans = max(ans, f[n][i]);
    cout << ans << endl;
    return 0;
}
```
