class Solution {
public:
    int countSubstrings(string s) {
        int ans = 0;
        for (int i = 0; i < s.size(); ++i) {
            for (int j = i, k = i; j >= 0 && k < s.size(); j--, k++) {
                if (s[j] != s[k]) break;
                ans++;
            }
            for (int j = i, k = i + 1; j >= 0 && k < s.size(); j--, k++) {
                if (s[j] != s[k]) break;
                ans++;
            }
        }
        return ans;
    }
};
